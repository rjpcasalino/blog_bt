i3 defaults to a status bar that has IPv6 and a bunch of other goodies. 

I'm more a "minimalist".

```
# i3status configuration file.
# see "man i3status" for documentation.

# It is important that this file is edited as UTF-8.
# The following line should contain a sharp s:
# ß
# If the above line is not correctly displayed, fix your editor first!

general {
  colors = true
  interval = 5
}

order += "ethernet enp21s0f0u2"
order += "wireless wlp59s0"
order += "disk /"
order += "cpu_temperature 0"
order += "cpu_usage"
order += "battery 0"
order += "tztime seattle"

battery 0 {
        format = "%status %percentage R:%remaining"
        format_down = "No battery"
        status_chr = "CHR"
        status_bat = "BAT"
        status_unk = "UNK"
        status_full = "FULL"
        path = "/sys/class/power_supply/BAT%d/uevent"
        low_threshold = 10
}

disk "/" {
  format = "/ %avail"
}

ethernet enp21s0f0u2 {
  format_up = "E: %ip"
  format_down = "E: down"
}

wireless wlp59s0 {
  format_up = "W: (%quality at %essid, %bitrate / %frequency) %ip"
  format_down = "W: down"
}

cpu_temperature 0 {
  format = "%degrees °C"
  path = "/sys/devices/platform/coretemp.0/hwmon/hwmon3/temp1_input"
}

tztime seattle {
        format = "%Y-%m-%d %H:%M %Z"
        timezone = "America/Los_Angeles"
}
```
More details can always be found in the manual: https://i3wm.org/i3status/manpage.html
